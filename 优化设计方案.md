### **2. 智能匹配算法**
```java
// 评分模型
public ShelterScore calculateScore(Shelter shelter, PetRequest request) {
    double distance = calculateDistance(
        shelter.getLat(), shelter.getLon(),
        request.getLat(), request.getLon()
    );
    double capacityRatio = (double)shelter.getCurrentCapacity() / shelter.getMaxCapacity();
    double medicalScore = shelter.getMedicalLevel() * 0.2;

    // 综合评分
    double score = (1.0 / (1.0 + distance/10)) * 0.4 +  // 距离权重40%
                   (1.0 - capacityRatio) * 0.3 +        // 容量权重30%
                   medicalScore * 0.3;                   // 医疗权重30%

    return new ShelterScore(shelter, score, distance);
}

// 容量管理
public String getCapacityAdvice(int current, int max) {
    double ratio = (double)current / max;
    if (ratio < 0.5) return "容量充足，推荐";
    else if (ratio < 0.7) return "容量适中，可接受";
    else if (ratio < 0.9) return "容量紧张，备选";
    else return "已满员，不推荐";
}
```

// 距离计算：Haversine公式（不用地图API）
public double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    double R = 6371.0; // 地球半径（公里）
    double dLat = Math.toRadians(lat2 - lat1);
    double dLon = Math.toRadians(lon2 - lon1);
    double a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
               Math.cos(Math.toRadians(lat1)) * Math.cos(Math.toRadians(lat2)) *
               Math.sin(dLon / 2) * Math.sin(dLon / 2);
    double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
}

// 容量管理：简单阈值判断
